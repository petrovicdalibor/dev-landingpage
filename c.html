Palindromi


#include
int obraten(n){
if(n/10==0){
return 0;
}
else{
return n*10+obraten(n/10);
}
}
int main(){
int n;
while(scanf("%d",&n)){
int tmp=n+obraten(n);
int pom=tmp;
int reverse=0;
while(pom!=0){
reverse*=10;
reverse+=pom%10;
pom/=10;
}
int obr=reverse;
reverse=0;
if(tmp==obr){
printf("Palindrom e: %d",n);
}
}
return 0;
}
